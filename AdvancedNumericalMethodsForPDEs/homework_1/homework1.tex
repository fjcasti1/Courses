\partial_tu(x,t)+a\partial_xu(x,t)-b\partial_x^2u(x,t)=0,~~~~
u(x,t=0)=u^I(x),
\end{align}
on the interval $x\in[-1, 1]$ with periodic boundary conditions $u(x + 2, t) = u(x, t), \forall x, t$.

\end{align}
