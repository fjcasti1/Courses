\begin{questions}

\question{For $n>0$, let
\begin{align*}
f_n(t)=1
\end{align*}
Show that $f_n\rightarrow 0$ in $L^2[0,1]$ but that $f_n(0)$ does not
converge to zero.
}
\begin{solution}
\begin{align*}
f_n(t)=\begin{cases} 
   \sqrt{n}, & 0\leq t\leq 1/n^2 \\
       ~~~0, & \text{otherwise}
\end{cases}
\end{align*}
Show that $f_n\rightarrow 0$ in $L^2[0,1]$ but that $f_n(0)$ does not converge to zero.
\end{solution}
\end{questions}
